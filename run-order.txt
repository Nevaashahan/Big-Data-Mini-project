1. Start Docker services
   docker compose up --build -d

2. Run Spring Boot
   cd springboot-app
   .\mvnw.cmd spring-boot:run

3. Run the producer
   cd producer
   pip install -r requirements.txt
   python clickstream_producer.py

4. Run the raw-events Spark job inside the Spark container
   docker exec ecommerce-spark-master sh -lc "mkdir -p /tmp/.ivy2 /tmp/checkpoints/raw-events && /opt/spark/bin/spark-submit --master local[*] --conf spark.jars.ivy=/tmp/.ivy2 --packages org.apache.spark:spark-sql-kafka-0-10_2.12:3.5.7 /opt/spark/jobs/raw_events_processor.py"

5. Run the metrics-and-alerts Spark job in a separate terminal
   docker exec ecommerce-spark-master sh -lc "mkdir -p /tmp/.ivy2 /tmp/checkpoints/metrics-alerts && /opt/spark/bin/spark-submit --master local[*] --conf spark.jars.ivy=/tmp/.ivy2 --packages org.apache.spark:spark-sql-kafka-0-10_2.12:3.5.7 /opt/spark/jobs/metrics_alert_processor.py"

6. Open Airflow and trigger the DAG
   URL: http://localhost:8083
   Login: admin / admin
   Optional demo conf for same-day data:
   {"report_date":"2026-04-16"}

7. Check Spring Boot endpoints
   http://localhost:8088/health
   http://localhost:8088/products
   http://localhost:8088/alerts
   http://localhost:8088/reports/top-products?reportDate=2026-04-16
   http://localhost:8088/reports/conversion?reportDate=2026-04-16
   http://localhost:8088/swagger-ui/index.html
